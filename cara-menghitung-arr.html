#include <iostream>

using namespace std;

int main () {

   int jumlah = 0;

   int totalitem = 8;

   int arr[totalitem] = {5, 2, 21, 4, 9, 16, 13, 3};

   for (int i = 0; i < totalitem; i++) {

       for (int j = 0; j < totalitem - i - 1; j++) {

           int x = j + 1;

           if (arr[j] > arr[x]) {

               int temp = arr[x];

               arr[x] = arr[j];

               arr[j] = temp;

           }

       }

   }

   for (int i = 0; i < totalitem; i++) {

       cout << arr[i] << " ";

   }

   for (int i = 0; i < totalitem; i++) {

       jumlah += arr[i];

   }

   cout << endl;

   cout << "Jumlah = " << jumlah << endl;

   return 0;

}
