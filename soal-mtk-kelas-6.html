1.

[tex] \frac{3}{8} + \frac{2}{5} = \frac{3 \times 5}{8 \times 5} + \frac{2 \times 8}{5 \times 8} [/tex]

[tex] = \frac{15}{40} + \frac{16}{40} [/tex]

[tex] = \frac{31}{40} [/tex]

2.

[tex] \frac{5}{12} + \frac{2}{3} = \frac{5}{12} + \frac{2 \times 4}{3 \times 4} [/tex]

[tex] = \frac{5}{12} + \frac{8}{12} [/tex]

[tex] = \frac{13}{12} [/tex]

[tex] = \frac{2 + 1}{13} [/tex]

[tex] = \frac{12}{12} + \frac{1}{12} [/tex]

[tex] = 1\frac{1}{12} [/tex]

3.

[tex] \frac{4}{9} + 3 \frac{2}{6} = \frac{4}{9} + \frac{20}{6} [/tex]

[tex] = \frac{8}{18} + \frac{60}{18} [/tex]

[tex] = \frac{68}{18} [/tex]

[tex] = 3 \frac{14}{18} [/tex]

4.

[tex] \frac{5}{8} + 1 \frac{3}{4}= \frac{5}{8} + \frac{7}{4} [/tex]

[tex]= \frac{5 \times 1}{8 \times 1} + \frac{7 \times 2}{4 \times 2} [/tex]

[tex] = \frac{5 + 14}{8} [/tex]

[tex] = \frac{19}{8} [/tex]

[tex] = 2\frac{3}{8} [/tex]

5.

[tex] \frac{10}{15} + 2 \frac{3}{5} = \frac{10}{15} + \frac{30}{5} [/tex]

[tex] = \frac{10 \times 1}{15 \times 1} + \frac{30 \times 3}{5 \times 3} [/tex]

[tex] = \frac{10}{15} + \frac{90}{15} [/tex]

[tex] = \frac{100}{15} [/tex]

[tex] = \frac{20}{3} [/tex]

[tex] = 6 \frac{2}{3} [/tex]

[tex] = 6,6[/tex]

= 22/7 × d

= 22/7 × 35

= 22 × 5

= 110 cm

= πr²

= 22/7 × 17,5²

= 22/7 × 306,25

= 22 × 43,75

= 962,5 cm²

= s²

= 35²

= 1.225 cm²
