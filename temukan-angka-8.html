Jawaban:

105

public class NO_1 {

static int binarySearch (int[]angka,int cari,int i, int j){

int index = -1;

while (i <= j){

int mid = (i+j)/2;

if (angka[mid] < cari){

i = mid + 1;

}

else if (angka[mid] > cari){

j = mid -1;

}

else if (angka [mid] == cari){

index = mid;

break;

}

}

return index;

}
