Pembahasan:

r = 66/44

[tex]\sf P = \frac {\alpha^{\circ}}{360^{\circ}} \times keliling~lingkaran[/tex]

[tex]\sf 22 = \frac {120^{\circ}}{360^{\circ}}\times \pi d[/tex]

[tex]\sf 22 = \frac {1}{3} \times \frac {22}{7} \times d[/tex]

[tex]\sf 22 = \frac {22}{21} \times d [/tex]

[tex]\sf 22 \times 21 = 22d [/tex]

[tex]\sf 462 = 22d[/tex]

[tex]\sf \frac {462}{22} = d[/tex]

[tex]\sf 21 = d[/tex]

•

[tex]\sf r = \frac {d}{2}[/tex]

[tex]\sf r = \frac {21}{2}[/tex]

[tex]\sf r = 10, 5~cm[/tex]
